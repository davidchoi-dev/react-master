# App 컴포넌트

클래스 컴포넌트로 작성된 App을 함수형 컴포넌트 & React 훅(Hooks)으로 변경하는 실습을 진행합니다. 클래스 컴포넌트에서만 사용되었던 상태(state), 라이프 사이클 훅(Life Cycle Hook)을 대체하는 React 훅을 활용해봅니다.

학습 목적으로 클래스 컴포넌트를 함수형 컴포넌트 & React 훅으로 바꿔보는 훈련을 하는 것일 뿐. 실제 프로젝트에서는 팀 또는 개인이 선호하는 방법을 사용하세요.

# 상태 → useState

useState() 훅을 사용하여 함수형 컴포넌트에서 상태를 관리할 수 있습니다.





 
 
 
 










import React, { useState } from 'react';

function App() {
  // 함수형 컴포넌트에서의 상태 관리
  const [state, setState] = useState({
    navigation: {},
    beverageList: []
  });

  return (
    <AppContext.Provider value={state}>
      <AppHeader />
      <AppMain />
      <GoToTop />
    </AppContext.Provider>
  )
}

# 라이프 사이클 훅 → useEffect

useEffect() 훅을 사용하여 함수형 컴포넌트에서 라이프 사이클 훅과 유사한 처리를 할 수 있습니다.

  • componentDidMount (초기 1회 실행)
  • componentDidUpdate (의존성 배열에 아이템이 추가된 경우)
  • componentWillUnmount (콜백 함수를 반환하는 경우)
import React, { useState, useEffect } from 'react';

// 핸들러 (비동기 함수 활용)
const fetchData = async () => {
  // try - catch 문
  try {
    // Firebase의 데이터 요청 후, 통신이 성공하면 데이터 추출
    const response = await fetch(FIREBASE_DB_ADDRESS);
    const { navigation, beverageList } = await response.json();
    // setState() 함수를 사용해 컴포넌트 상태 업데이트
    setState({ navigation, beverageList });
  } 
  // 통신 과정에서 오류가 발생한 경우
  catch (error) {
    // 오류 메시지 출력
    console.error(error.message);
  }
};

// 라이프 사이클 훅을 대체하는 사이드 이펙트 훅
useEffect(
  () => {
    // 데이터 패치 함수 실행
    fetchData();
  }, 
  // 의존 없음 (componentDidMount, 1회 실행)
  []
);

# 성능 저하 방지 → useCallback

useCallback() 훅을 사용하여 컴포넌트가 업데이트 될 때 마다 불필요한 과정이 발생하지 않도록 설정합니다.

import React, { useState, useEffect, useCallback } from 'react';

// 핸들러 (useCallback 훅 활용)
const fetchData = useCallback(
  // 비동기 함수 활용
  async () => {
    // try - catch 문
    try {
      // Firebase의 데이터 요청 후, 통신이 성공하면 데이터 추출
      const response = await fetch(FIREBASE_DB_ADDRESS);
      const { navigation, beverageList } = await response.json();
      // setState() 함수를 사용해 컴포넌트 상태 업데이트
      setState({ navigation, beverageList });
    } 
    // 통신 과정에서 오류가 발생한 경우
    catch (error) {
      // 오류 메시지 출력
      console.error(error.message);
    }
  }, 
  // 의존 없음 (componentDidMount, 1회 실행)
  []
);

// 라이프 사이클 훅을 대체하는 사이드 이펙트 훅
useEffect(
  () => {
    // 데이터 패치 함수 실행
    fetchData();
  }, 
  // fetchData 의존 (componentDidMount, 1회 실행)
  [fetchData]
);
