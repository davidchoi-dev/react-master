# GoToTop 컴포넌트

학습 목적에 따라 GoToTop 컴포넌트를 클래스 컴포넌트에서 함수형 컴포넌트로 변경해봅니다. 이미 앞서 다룬 튜토리얼에서 많이 다뤘던 주제를 반복하는 것이므로 여기서는 완성된 코드를 보고 넘어가겠습니다.

GoToTop.scss 파일을 제거하고, 스타일 컴포넌트로 만드는 부분은 생략했습니다. 하지만, 여러분들은 실습해보길 바랍니다. 😄

import './GoToTop.scss';
import React, { useState, useCallback, useEffect } from 'react';
import classNames from 'classnames';
import AppButton from '@components/AppButton/AppButton';

/**
 * GoToTop 컴포넌트
 */
function GoToTop(props) {
  const [ isShow, setIsShow ] = useState(false);

  const handleScrollDetection = useCallback(
    () => {
      window.scrollY > props.changeModePosition
        ? setIsShow(true)
        : setIsShow(false);
    },
    [ props.changeModePosition ]
  );

  useEffect(
    () => {
      window.addEventListener('scroll', handleScrollDetection);
      return () => {
        window.removeEventListener('scroll', handleScrollDetection);
      };
    },
    [ handleScrollDetection ]
  );

  // 클래스 속성 병합
  const combineClassNaems = classNames('goToTop', isShow ? 'is-show' : '');

  return (
    <AppButton
      aria-label="페이지 상단 이동"
      className={combineClassNaems}
      onClick={GoToTop.handleGoToTop}
    >
      <svg
        aria-hidden="true"
        viewBox="0 0 64 64"
        xmlns="http://www.w3.org/2000/svg"
      >
        <path d="m32 56c1.104 0 2-.896 2-2v-39.899l14.552 15.278c.393.413.92.621 1.448.621.495 0 .992-.183 1.379-.552.8-.762.831-2.028.069-2.828l-16.619-17.448c-.756-.755-1.76-1.172-2.829-1.172s-2.073.417-2.862 1.207l-16.586 17.414c-.762.8-.731 2.066.069 2.828s2.067.731 2.828-.069l14.551-15.342v39.962c0 1.104.896 2 2 2z" />
      </svg>
    </AppButton>
  );
}

GoToTop.defaultProps = {
  changeModePosition: 100,
};

GoToTop.handleGoToTop = () => {
  document.documentElement.scrollIntoView({ behavior: 'smooth' });
};

export default GoToTop;
