# BeverageItem 컴포넌트

학습 목적에 따라 BeverageItem 컴포넌트를 클래스 컴포넌트에서 함수형 컴포넌트로 변경해봅니다.

# 사용할 React 훅

이번 학습에 사용할 React 훅은 2가지 입니다.

  • 컴포넌트 상태: useState
  • 컴포넌트 이벤트 핸들러: useCallback
import React, { Fragment, useState, useCallback } from 'react';

# 상수 설정

다이얼로그 클래스 이름은 코드 로직 안에서 자주 사용되므로 상수로 함수형 컴포넌트 외부에 설정합니다.

const DIALOG_CLASSNAME = 'beverageList__dialog';
1

# 상태 설정

AppNavigation 컴포넌트에서 총체적으로 React 훅을 다뤄봤고, BeverageItem 컴포넌트 또한 유사한 로직을 다룹니다. 시간 차를 두고 상태를 나눠 업데이트(애니메이션) 해야 하므로, 여기서는 컴포넌트 상태 통합 관리 보다 개별 관리가 더 나은 선택입니다.

const [ isPressed, setPressed ] = useState(false);
const [ dialogClassName, setDialogClassName ] = useState(DIALOG_CLASSNAME);
1
2

# 이벤트 핸들러 설정

AppNavigation 컴포넌트에서 다뤘던 로직과 유사합니다. 각 핸들러 마다 시간차를 두고 상태를 개별 업데이트 하도록 수정합니다.

const handleShowDialog = useCallback((e) => {
  e.preventDefault();
  setPressed(true);
  window.setTimeout(
    () => setDialogClassName(`${DIALOG_CLASSNAME} is-active`),
    100
  );
}, []);

const handleHideDialog = useCallback(() => {
  setDialogClassName(DIALOG_CLASSNAME);
  window.setTimeout(() => setPressed(false), 400);
}, []);

# 다이얼로그 ID 설정

각 다이얼로그는 충돌 없는 고유한 ID 설정이 필요합니다. 앞서 정의한 다이얼로그 클래스 이름 상수 값에 전달 받은 아이템의 ID 문자 값을 병합하여 고유한 ID 값을 설정한 변수를 정의합니다.

const uniqueId = `${DIALOG_CLASSNAME}-${id}`;

# 렌더링 설정

클래스 컴포넌트가 아니므로 this 키워드를 사용했던 이전 코드를 함수 컴포넌트 안에서 접근 가능한 상태 변수, 핸들러 함수로 연결 수정합니다. 함수형 컴포넌트 업데이트 마다 핸들러가 다시 렌더링 과정에서 사용되지 않도록 useCallback 훅을 사용해 성능 문제를 해결합니다.








 
 








 
























 







return (
  <li className="beverageList__item">
    <AppButton
      className="beverageList__button"
      as="a"
      href="#"
      aria-haspopup="true"
      aria-pressed={isPressed}
      onClick={handleShowDialog}

        <img src={src} width={width} height={height} alt="" />
        <figcaption>{name}</figcaption>
      </figure>

    <div
      className={dialogClassName}

      aria-modal="false"
      aria-labelledby={uniqueId}

      </h3>
      <p>{desc}</p>
      <div className="beverageList__dialog--multiColumn is-2">
        <dl>
          {criteria.map(([ dt, dd ], index) => (
            <Fragment key={index}>
              <dt>{dt}</dt>
              <dd>({dd})</dd>
            </Fragment>
          ))}
        </dl>
      </div>
      <button
        className="resetButton is-close-dialog"
        type="button"
        title="닫기"
        aria-label="음료 정보 패널 닫기"
