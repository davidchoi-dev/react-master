# useFetch 커스텀 훅

App 컴포넌트에서 작성했던 useEffect 훅과 Fetch API 코드는 비동기 통신 요청할 때마다 자주 사용되는 코드입니다. 앞으로도 비동기 통신 요청이 필요할 때 마다 유사한 코드를 작성해야겠죠. 하지만 보다시피 작성해야 할 코드 양이 많아 복사/붙여넣기 할 경우가 많을 것입니다. 번거롭고 수정 사항이 발생하면 관리하기도 쉽지 않을 것 같습니다.

// 상태
const [state, setState] = useState({
  navigation: {},
  beverageList: []
});

// 라이프 사이클 훅 대체
useEffect(
  () => {
    // 비동기 함수 (비동기 통신 요청)
    async function fetchData() {
      try {
        // 응답 받은 데이터
        const response = await fetch(FIREBASE_DB_ADDRESS);
        // JSON → 필요한 데이터 추출
        const { navigation, beverageList } = await response.json();
        // 상태 업데이트
        setState({
          navigation,
          beverageList,
        });
      } 
      // 오류 발생 시, 처리
      catch (error) {
        console.error(error.message);
      }
    }
    // 데이터 패치 요청 함수 실행
    fetchData();
  }, 
  // 의존 없음
  []
);

# useFetch 훅 생성

매번 중복되는 코드 로직을 분리하여 재사용 할 수 있다면 사용과 관리에 매우 유용할 것입니다. useEffect 훅과 Fetch API를 묶어 재사용 할 수 있는 커스텀 훅을 생성해 활용해봅시다. (useState, useEffect 훅 / Fetch API 활용)

src/hooks/useFetch.js

/* eslint-disable react-hooks/exhaustive-deps */
import { useState, useEffect } from 'react';

function useFetch(endpoint, options = {}) {
  // 상태
  const [ value, setValue ] = useState(null);
  const [ loading, setLoading ] = useState(false);
  const [ error, setError ] = useState(false);

  // 사이드 이펙트
  useEffect(
    () => {
      // 데이터 패치 함수
      const fetchData = async () => {
        setLoading(true);
        try {
          const response = await fetch(endpoint, options);
          const json = await response.json();
          setValue(json);
        } catch (error) {
          setError(error);
        }
        setLoading(false);
      };

      // 데이터 패치 함수 실행
      fetchData();
    }, 
    // 의존 없음
    []
  );

  // 반환 (값)
  return [ value, loading, error ];
}

export default useFetch;

# useFetch 훅 활용

정의한 useFetch 훅을 App 컴포넌트에서 불러들여 컴포넌트 내부에서 활용해봅니다. useFetch 훅을 통해 처리된 결과 값이 비동기 통신 후, 반환되면 useEffect 훅에서 이를 감지하여 컴포넌트 상태를 업데이트 하도록 코드를 다음과 같이 작성합니다.

src/App/App.js

import useFetch from '~/hooks/useFetch';

/**
 * App 컴포넌트
 */
function App() {
  // 생략......

  // useFetch 훅 활용 (코드 양 감소)
  const [ value, loading, error ] = useFetch(FIREBASE_DB_ADDRESS);

  // useFetch 훅의 결과 값 비동기 처리 감지 후, 실행되는 useEffect 훅
  useEffect(
    () => {
      if (value) {
        const { navigation, beverageList } = value;
        setState({
          navigation,
          beverageList
        });
      }
    },
    [ value ]
  );

  // 로딩 중일 경우, 로딩 스피너 표시
  if (loading) {
    return <AppLoading />;
  }

  // 오류가 발생하면? 오류 콘텐츠 표시
  if (error) {
    return <div>오류 발생!</div>;
  }

  // 렌더링
  return (
    // 생략......
  );
}

훅을 사용했는데 오류가 발생하나요?

훅 활용 규칙을 어길 경우, 훅이 정상 작동하지 않으니 유의해야 합니다.

  • 일반 함수, 클래스 컴포넌트 안에서 훅을 사용할 수 없다.
  • 오직 함수형 컴포넌트 안에서만 훅을 사용할 수 있다.
  • 조건문, 반복문, 중복 함수 안에서 훅을 사용할 수 없다.
